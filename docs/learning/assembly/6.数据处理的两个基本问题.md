## 使用变址寄存器si和di
## 更灵活的寻址方式
1. `[bx+idata]`：[bx+3]
2. `[bx+si]`：[bx+si]
3. `[bx+di+idata]`：[bx+di+idata]

```
assume cs:code,ds:data,ss:stack
code segment
    start:
        mov ax,data
        mov ds,ax
        mov bx,0H
        mov ax,stack
        mov ss,ax
        mov sp,1
        mov cx,4
        o:
            push cx
            mov cx,3
            mov si,3
            i:
                mov al,[bx+si]
                and al,1011111B
                mov [bx+si],al
                inc si
            loop i

            pop cx
            add bx,16
        loop o

        mov ax,4c00h
        int 21h
code ends

data segment
    db '1. hello        '
    db '2. brows        '
    db '3. replace      '
    db '4. modify       '
data ends

stack segment
dw 0
stack ends

end
```
